# Dependencias para el bot de Telegram
python-telegram-bot

# Para cargar las variables de entorno desde el archivo .env
python-dotenv

# Librería oficial de OpenAI, usada para la compatibilidad con la API de OpenRouter
openai

# (Opcional, pero muy recomendado) Para manejar imágenes fácilmente
# Pillow es necesaria para algunas operaciones con imágenes, aunque a veces 
# python-telegram-bot la instala como dependencia. Es bueno tenerla explícitamente.
Pillow

# Dependencias para el servidor web FastAPI
fastapi
uvicorn[standard]

# Para manejar CORS (Cross-Origin Resource Sharing) - necesario para comunicación web
fastapi-cors

# Para validación de datos robusta con Pydantic
pydantic

# Para hacer peticiones HTTP a OpenRouter
httpx

# Para descargar imágenes desde URLs
requests
